# Build Xen
Install dependency:
```
sudo apt update && sudo apt install -y \
    build-essential \
    gcc \
    g++ \
    clang \
    binutils \
    make \
    gawk \
    zlib1g-dev \
    python3-dev \
    libncurses-dev \
    uuid-dev \
    libyajl-dev \
    libaio-dev \
    libglib2.0-dev \
    libpixman-1-dev \
    pkg-config \
    bridge-utils \
    iproute2 \
    bison \
    flex \
    acpica-tools
```
